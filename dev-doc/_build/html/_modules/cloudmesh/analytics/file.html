[docs]def save(file):
    """
    Save file after securing the file name
    :param file:
    :return:
    """
    filename = secure_filename(file.filename)
    try:
        file.save(os.path.join(current_app.config['UPLOAD_FOLDER'], filename))
    except Exception as e:
        return jsonify({'error_message': str(e)})
